61template <typename T, size_t N, typename DensityFn>
62std::vector<Point<T, N>> rejection_sampling(
63 size_t count,
64 const std::array<std::pair<T, T>, N> &axis_ranges,
65 DensityFn density_fn,
66 std::optional<unsigned int> seed = std::nullopt)
67{
68 std::mt19937 gen(seed ? *seed : std::random_device{}());
69
70 std::array<std::uniform_real_distribution<T>, N> coord_dists;
71 for (size_t i = 0; i < N; ++i)
72 coord_dists[i] = std::uniform_real_distribution<T>(axis_ranges[i].first,
73 axis_ranges[i].second);
74
75 std::vector<Point<T, N>> candidates;
76 candidates.reserve(count * 2); // overgenerate
77
78 while (candidates.size() < count * 2)
79 {
80 Point<T, N> p;
81 for (size_t i = 0; i < N; ++i)
82 p[i] = coord_dists[i](gen);
83
84 candidates.push_back(p);
85 }
86
87 return function_rejection_filter<T, N>(candidates, density_fn, seed);
88}
