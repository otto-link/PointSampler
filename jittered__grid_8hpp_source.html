45template <typename T, size_t N>
46std::vector<Point<T, N>> jittered_grid(size_t count,
47 const std::array<std::pair<T, T>, N> &axis_ranges,
48 const std::array<T, N> &jitter_amount,
49 const std::array<T, N> &stagger_ratio,
50 std::optional<unsigned int> seed = std::nullopt)
51{
52 std::mt19937 gen(seed ? *seed : std::random_device{}());
53 std::uniform_real_distribution<T> uniform01(0.0, 1.0);
54
55 std::array<size_t, N> resolution;
56 size_t total_cells = 1;
57
58 T volume = 1;
59 for (const auto &[min, max] : axis_ranges)
60 volume *= (max - min);
61
62 T target_cell_volume = volume / static_cast<T>(count);
63 T cell_size_estimate = std::pow(target_cell_volume, static_cast<T>(1.0) / N);
64
65 for (size_t i = 0; i < N; ++i)
66 {
67 T range = axis_ranges[i].second - axis_ranges[i].first;
68 resolution[i] = std::max<size_t>(1, static_cast<size_t>(range / cell_size_estimate));
69 total_cells *= resolution[i];
70 }
71
72 std::vector<Point<T, N>> points;
73 points.reserve(std::min(count, total_cells));
74
75 std::vector<std::array<size_t, N>> cell_indices;
76 for (size_t linear = 0; linear < total_cells; ++linear)
77 {
78 std::array<size_t, N> index{};
79 size_t div = 1;
80 for (size_t i = 0; i < N; ++i)
81 {
82 index[i] = (linear / div) % resolution[i];
83 div *= resolution[i];
84 }
85 cell_indices.push_back(index);
86 }
87
88 std::shuffle(cell_indices.begin(), cell_indices.end(), gen);
89 size_t limit = std::min(count, cell_indices.size());
90
91 for (size_t i = 0; i < limit; ++i)
92 {
93 const auto &idx = cell_indices[i];
94 Point<T, N> p;
95
96 for (size_t d = 0; d < N; ++d)
97 {
98 T range_min = axis_ranges[d].first;
99 T range_max = axis_ranges[d].second;
100 T cell_size = (range_max - range_min) / static_cast<T>(resolution[d]);
101
102 T jitter_range = jitter_amount[d] * cell_size;
103 T jitter_center = (1.0 - jitter_amount[d]) * 0.5 * cell_size;
104 T jitter = uniform01(gen) * jitter_range;
105
106 // Compute stagger offset from higher dimensions
107 T stagger_offset = 0;
108 for (size_t k = d + 1; k < N; ++k)
109 {
110 if (idx[k] % 2 == 1)
111 stagger_offset += stagger_ratio[d] * cell_size;
112 }
113
114 p[d] = range_min + idx[d] * cell_size + jitter_center + jitter + stagger_offset;
115 }
116
117 points.push_back(p);
118 }
119
120 return points;
121}

143template <typename T, size_t N>
