34template <typename T, std::size_t N>
35std::vector<Point<T, N>> random_rejection_filter(const std::vector<Point<T, N>> &points,
36 std::size_t target_count)
37{
38 if (target_count >= points.size())
39 {
40 return points; // nothing to remove
41 }
42
43 std::vector<std::size_t> indices(points.size());
44 std::iota(indices.begin(), indices.end(), 0);
45
46 std::random_device rd;
47 std::mt19937 gen(rd());
48 std::shuffle(indices.begin(), indices.end(), gen);
49
50 // Take the first `target_count` shuffled indices
51 std::vector<Point<T, N>> result;
52 result.reserve(target_count);
53 for (std::size_t i = 0; i < target_count; ++i)
54 {
55 result.push_back(points[indices[i]]);
56 }
57
58 return result;
59}
