34template <typename T, size_t N>
35std::vector<Point<T, N>> filter_points_in_range(
36 const std::vector<Point<T, N>> &points,
37 const std::array<std::pair<T, T>, N> &axis_ranges)
38{
39 std::vector<Point<T, N>> filtered;
40 filtered.reserve(points.size());
41
42 for (const auto &p : points)
43 {
44 bool inside = true;
45 for (size_t i = 0; i < N; ++i)
46 {
47 const auto &[min_val, max_val] = axis_ranges[i];
48 if (p[i] < min_val || p[i] > max_val)
49 {
50 inside = false;
51 break;
52 }
53 }
54 if (inside)
55 filtered.push_back(p);
56 }
57
58 return filtered;
59}

123template <typename T, size_t N>
124void refit_points_to_range(std::vector<Point<T, N>> &points,
125 const std::array<std::pair<T, T>, N> &target_ranges)
126{
127 if (points.empty())
128 return;
129
130 std::array<T, N> min_vals, max_vals;
131
132 // Initialize min/max
133 for (size_t d = 0; d < N; ++d)
134 {
135 min_vals[d] = points[0][d];
136 max_vals[d] = points[0][d];
137 }
138
139 // Compute bounding box
140 for (const auto &p : points)
141 {
142 for (size_t d = 0; d < N; ++d)
143 {
144 min_vals[d] = std::min(min_vals[d], p[d]);
145 max_vals[d] = std::max(max_vals[d], p[d]);
146 }
147 }
148
149 // Apply linear mapping to each point
150 for (auto &p : points)
151 {
152 for (size_t d = 0; d < N; ++d)
153 {
154 const T in_min = min_vals[d];
155 const T in_max = max_vals[d];
156 const T out_min = target_ranges[d].first;
157 const T out_max = target_ranges[d].second;
158
159 if (std::abs(in_max - in_min) < T(1e-12))
160 {
161 // Degenerate axis: center the target range
162 p[d] = (out_min + out_max) / T(2);
163 }
164 else
165 {
166 const T t = (p[d] - in_min) / (in_max - in_min);
167 p[d] = out_min + t * (out_max - out_min);
168 }
169 }
170 }
171}

193template <typename T, size_t N>
194void rescale_points(std::vector<Point<T, N>> &points,
195 const std::array<std::pair<T, T>, N> &ranges)
196{
197 for (auto &pt : points)
198 for (size_t d = 0; d < N; ++d)
199 pt[d] = ranges[d].first + pt[d] * (ranges[d].second - ranges[d].first);
200}
