52template <typename T, size_t N, typename DensityFn>
53std::vector<Point<T, N>> importance_resampling(
54 size_t count,
55 size_t oversampling_ratio,
56 const std::array<std::pair<T, T>, N> &axis_ranges,
57 DensityFn density_fn,
58 std::optional<unsigned int> seed = std::nullopt)
59{
60 std::mt19937 gen(seed ? *seed : std::random_device{}());
61
62 // Generate grid
63 size_t count_grid = count * oversampling_ratio;
64 std::vector<Point<T, N>> grid_points = halton<T, N>(count_grid, axis_ranges, seed);
65
66 // Weights
67 std::vector<T> weights;
68 weights.reserve(count_grid);
69
70 for (auto &p : grid_points)
71 weights.push_back(density_fn(p));
72
73 // Normalize weights
74 T sum = std::accumulate(weights.begin(), weights.end(), T(0));
75 for (auto &w : weights)
76 w /= sum;
77
78 // Resample
79 std::discrete_distribution<size_t> dist(weights.begin(), weights.end());
80 std::vector<Point<T, N>> samples;
81 samples.reserve(count);
82 for (size_t i = 0; i < count; ++i)
83 samples.push_back(grid_points[dist(gen)]);
84
85 return samples;
86}
