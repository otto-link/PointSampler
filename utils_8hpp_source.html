140template <typename T, size_t N>
141std::vector<Point<T, N + 1>> add_dimension(const std::vector<Point<T, N>> &points,
142 const std::vector<T> &new_dimension)
143{
144 if (points.size() != new_dimension.size())
145 throw std::runtime_error(
146 "add_dimension: size mismatch between points and new dimension data");
147
148 std::vector<Point<T, N + 1>> result;
149 result.reserve(points.size());
150
151 for (size_t i = 0; i < points.size(); ++i)
152 {
153 std::array<T, N + 1> coords{};
154 std::copy(points[i].coords.begin(), points[i].coords.end(), coords.begin());
155 coords[N] = new_dimension[i];
156 result.emplace_back(coords);
157 }
158
159 return result;
160}

198template <typename T, size_t N>
199std::vector<Point<T, N>> merge_by_dimension(
200 const std::array<std::vector<T>, N> &components)
201{
202 if constexpr (N > 0)
203 {
204 std::size_t count = components[0].size();
205 for (std::size_t i = 1; i < N; ++i)
206 {
207 if (components[i].size() != count)
208 throw std::invalid_argument("All component vectors must have the same size");
209 }
210
211 std::vector<Point<T, N>> points(count);
212 for (std::size_t i = 0; i < count; ++i)
213 {
214 for (std::size_t j = 0; j < N; ++j)
215 points[i][j] = components[j][i];
216 }
217
218 return points;
219 }
220 else
221 {
222 return {};
223 }

251template <typename T, size_t N> void normalize_points(std::vector<Point<T, N>> &points)
252{
253 if (points.empty())
254 return;
255
256 std::array<T, N> min_vals;
257 std::array<T, N> max_vals;
258
259 // Initialize min/max arrays
260 for (size_t dim = 0; dim < N; ++dim)
261 {
262 min_vals[dim] = points[0].coords[dim];
263 max_vals[dim] = points[0].coords[dim];
264 }
265
266 // Find min/max for each axis
267 for (const auto &p : points)
268 {
269 for (size_t dim = 0; dim < N; ++dim)
270 {
271 min_vals[dim] = std::min(min_vals[dim], p.coords[dim]);
272 max_vals[dim] = std::max(max_vals[dim], p.coords[dim]);
273 }
274 }
275
276 // Normalize each point
277 for (auto &p : points)
278 {
279 for (size_t dim = 0; dim < N; ++dim)
280 {
281 T range = max_vals[dim] - min_vals[dim];
282 if (range != T(0))
283 p.coords[dim] = (p.coords[dim] - min_vals[dim]) / range;
284 else
285 p.coords[dim] = T(0); // Avoid NaN if all values
286 // are equal
287 }
288 }
289}
