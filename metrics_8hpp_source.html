41template <typename T, size_t N>
42std::vector<T> distance_to_boundary(const std::vector<Point<T, N>> &points,
43 const std::array<std::pair<T, T>, N> &axis_ranges)
44{
45 std::vector<T> distances;
46 distances.reserve(points.size());
47
48 for (const auto &p : points)
49 {
50 // Find min distance to a boundary plane
51 T min_dist = std::numeric_limits<T>::max();
52
53 for (size_t d = 0; d < N; ++d)
54 {
55 T dist_to_min = std::abs(p[d] - axis_ranges[d].first);
56 T dist_to_max = std::abs(axis_ranges[d].second - p[d]);
57
58 min_dist = std::min({min_dist, dist_to_min, dist_to_max});
59 }
60
61 distances.push_back(min_dist);
62 }
63
64 return distances;
65}

106template <typename T, size_t N>
107std::vector<T> first_neighbor_distance_squared(std::vector<Point<T, N>> &points)
108{
109 PointCloudAdaptor<T, N> adaptor(points);
110 KDTree<T, N> index(N, adaptor);
111 index.buildIndex();
112
113 std::vector<T> distance_sq;
114 distance_sq.reserve(points.size());
115
116 // first neighbor search only
117 const size_t k_neighbors = 1;
118
119 for (size_t i = 0; i < points.size(); ++i)
120 {
121 const auto &p = points[i];
122
123 std::vector<size_t> ret_indexes(k_neighbors + 1);
124 std::vector<T> out_dists_sqr(k_neighbors + 1);
125
126 nanoflann::KNNResultSet<T> result_set(k_neighbors + 1);
127 result_set.init(ret_indexes.data(), out_dists_sqr.data());
128
129 std::array<T, N> query;
130 for (size_t d = 0; d < N; ++d)
131 query[d] = p[d];
132
133 index.findNeighbors(result_set, query.data(), nanoflann::SearchParameters());
134
135 T dist_sq = 0.f;
136 if (ret_indexes.size() > 1)
137 {
138 const auto &q = points[ret_indexes[1]];
139 auto delta = p - q;
140 dist_sq = length_squared(delta) + T(1e-6);
141 }
142
143 distance_sq.push_back(dist_sq);
144 }
145
146 return distance_sq;
147}

183 const std::vector<Point<T, N>> &points,
184 size_t k_neighbors = 8)

188 index.buildIndex();
189

191 all_neighbors.resize(points.size());

196
197 std::vector<size_t> ret_indexes(k_neighbors + 1);
198 std::vector<T> out_dists_sqr(k_neighbors + 1);
199
200 nanoflann::KNNResultSet<T> result_set(k_neighbors + 1);
201 result_set.init(ret_indexes.data(), out_dists_sqr.data());
202
203 std::array<T, N> query;
204 for (size_t d = 0; d < N; ++d)
205 query[d] = p[d];
206
207 index.findNeighbors(result_set, query.data(), nanoflann::SearchParameters());
208
209 // Skip self at index 0
210 all_neighbors[i].assign(ret_indexes.begin() + 1, ret_indexes.end());
211 }

255template <typename T, size_t N>
256std::pair<std::vector<T>, std::vector<T>> radial_distribution(
257 const std::vector<Point<T, N>> &points,
258 const std::array<std::pair<T, T>, N> &axis_ranges,
259 T bin_width,
260 T max_distance)
261{
262 size_t num_bins = static_cast<size_t>(std::ceil(max_distance / bin_width));
263 std::vector<T> radii(num_bins);
264 std::vector<T> g(num_bins, T(0));
265
266 // Precompute bin centers
267 for (size_t i = 0; i < num_bins; ++i)
268 radii[i] = (i + T(0.5)) * bin_width;
269
270 // Compute volume of domain
271 T volume = T(1);
272 for (const auto &range : axis_ranges)
273 volume *= (range.second - range.first);
274
275 size_t n_points = points.size();
276 T density = static_cast<T>(n_points) / volume;
277
278 // Histogram of pair distances
279 for (size_t i = 0; i < n_points; ++i)
280 {
281 for (size_t j = i + 1; j < n_points; ++j)
282 {
283 T dist = distance(points[i], points[j]);
284 if (dist < max_distance)
285 {
286 size_t bin = static_cast<size_t>(dist / bin_width);
287 if (bin < num_bins)
288 g[bin] += T(2); // count both i→j and
289 // j→i
290 }
291 }
292 }
293
294 // Normalize
295 T norm_factor = (T)n_points * density;
296
297 // Volume of spherical shell between r1 and r2 in N dimensions
298 auto sphere_volume = [](T radius)
299 { return std::pow(M_PI, N / 2.0) / std::tgamma(N / 2.0 + 1.0) * std::pow(radius, N); };
300
301 for (size_t i = 0; i < num_bins; ++i)
302 {
303 T r_inner = i * bin_width;
304 T r_outer = (i + 1) * bin_width;
305
306 // Volume of spherical shell
307 T shell_vol = sphere_volume(r_outer) - sphere_volume(r_inner);
308 g[i] /= norm_factor * shell_vol;
309 }
310
311 return {radii, g};
312}
