47template <typename T, size_t N>
48std::pair<std::vector<T>, std::vector<T>> angle_distribution_neighbors(
49 const std::vector<Point<T, N>> &points,
50 T bin_width,
51 size_t k_neighbors = 8)
52{
53 if constexpr (N < 2)
54 throw std::runtime_error("Angle distribution requires dimension >= 2");
55
56 size_t num_bins = static_cast<size_t>(std::ceil(M_PI / bin_width));
57 std::vector<T> angles(num_bins);
58 std::vector<T> g_theta(num_bins, T(0));
59
60 // Precompute bin centers
61 for (size_t i = 0; i < num_bins; ++i)
62 angles[i] = (i + T(0.5)) * bin_width;
63
64 // Get nearest neighbors once
65 auto neighbors = nearest_neighbors_indices(points, k_neighbors);
66
67 // Loop over all points
68 for (size_t i = 0; i < points.size(); ++i)
69 {
70 const auto &nbrs = neighbors[i];
71 size_t n_nbrs = nbrs.size();
72
73 // Compute angles between all neighbor pairs
74 for (size_t a = 0; a < n_nbrs; ++a)
75 {
76 for (size_t b = a + 1; b < n_nbrs; ++b)
77 {
78 size_t j = nbrs[a];
79 size_t k = nbrs[b];
80
81 // Vectors from center i to j and k
82 std::array<T, N> v1, v2;
83 for (size_t d = 0; d < N; ++d)
84 {
85 v1[d] = points[j][d] - points[i][d];
86 v2[d] = points[k][d] - points[i][d];
87 }
88
89 // Compute angle
90 T dot = T(0), norm1 = T(0), norm2 = T(0);
91 for (size_t d = 0; d < N; ++d)
92 {
93 dot += v1[d] * v2[d];
94 norm1 += v1[d] * v1[d];
95 norm2 += v2[d] * v2[d];
96 }
97
98 if (norm1 > T(0) && norm2 > T(0))
99 {
100 T cos_theta = std::clamp(dot / (std::sqrt(norm1) * std::sqrt(norm2)),
101 T(-1),
102 T(1));
103 T theta = std::acos(cos_theta);
104
105 size_t bin = static_cast<size_t>(theta / bin_width);
106 if (bin < num_bins)
107 g_theta[bin] += T(1);
108 }
109 }
110 }
111 }
112
113 // Normalize histogram
114 T total = std::accumulate(g_theta.begin(), g_theta.end(), T(0));
115 if (total > T(0))
116 for (auto &val : g_theta)
117 val /= total;
118
119 return {angles, g_theta};
120}

149template <typename T, size_t N>
150std::vector<T> distance_to_boundary(const std::vector<Point<T, N>> &points,
151 const std::array<std::pair<T, T>, N> &axis_ranges)
152{
153 std::vector<T> distances;
154 distances.reserve(points.size());
155
156 for (const auto &p : points)
157 {
158 // Find min distance to a boundary plane
159 T min_dist = std::numeric_limits<T>::max();
160
161 for (size_t d = 0; d < N; ++d)
162 {
163 T dist_to_min = std::abs(p[d] - axis_ranges[d].first);
164 T dist_to_max = std::abs(axis_ranges[d].second - p[d]);
165
166 min_dist = std::min({min_dist, dist_to_min, dist_to_max});
167 }
168
169 distances.push_back(min_dist);
170 }
171
172 return distances;
173}

214template <typename T, size_t N>
215std::vector<T> first_neighbor_distance_squared(std::vector<Point<T, N>> &points)
216{
217 PointCloudAdaptor<T, N> adaptor(points);
218 KDTree<T, N> index(N, adaptor);
219 index.buildIndex();
220
221 std::vector<T> distance_sq;
222 distance_sq.reserve(points.size());
223
224 // first neighbor search only
225 const size_t k_neighbors = 1;
226
227 for (size_t i = 0; i < points.size(); ++i)
228 {
229 const auto &p = points[i];
230
231 std::vector<size_t> ret_indexes(k_neighbors + 1);
232 std::vector<T> out_dists_sqr(k_neighbors + 1);
233
234 nanoflann::KNNResultSet<T> result_set(k_neighbors + 1);
235 result_set.init(ret_indexes.data(), out_dists_sqr.data());
236
237 std::array<T, N> query;
238 for (size_t d = 0; d < N; ++d)
239 query[d] = p[d];
240
241 index.findNeighbors(result_set, query.data(), nanoflann::SearchParameters());
242
243 T dist_sq = 0.f;
244 if (ret_indexes.size() > 1)
245 {
246 const auto &q = points[ret_indexes[1]];
247 auto delta = p - q;
248 dist_sq = length_squared(delta) + T(1e-6);
249 }
250
251 distance_sq.push_back(dist_sq);
252 }
253
254 return distance_sq;
255}

289template <typename T, size_t N>
290std::vector<std::vector<size_t>> nearest_neighbors_indices(
291 const std::vector<Point<T, N>> &points,
292 size_t k_neighbors = 8)
293{
294 PointCloudAdaptor<T, N> adaptor(points);
295 KDTree<T, N> index(N, adaptor);
296 index.buildIndex();
297
298 std::vector<std::vector<size_t>> all_neighbors;
299 all_neighbors.resize(points.size());
300
301 for (size_t i = 0; i < points.size(); ++i)
302 {
303 const auto &p = points[i];
304
305 std::vector<size_t> ret_indexes(k_neighbors + 1);
306 std::vector<T> out_dists_sqr(k_neighbors + 1);
307
308 nanoflann::KNNResultSet<T> result_set(k_neighbors + 1);
309 result_set.init(ret_indexes.data(), out_dists_sqr.data());
310
311 std::array<T, N> query;
312 for (size_t d = 0; d < N; ++d)
313 query[d] = p[d];
314
315 index.findNeighbors(result_set, query.data(), nanoflann::SearchParameters());
316
317 // Skip self at index 0
318 all_neighbors[i].assign(ret_indexes.begin() + 1, ret_indexes.end());
319 }
320
321 return all_neighbors;
322}

363template <typename T, size_t N>
364std::pair<std::vector<T>, std::vector<T>> radial_distribution(
365 const std::vector<Point<T, N>> &points,
366 const std::array<std::pair<T, T>, N> &axis_ranges,
367 T bin_width,
368 T max_distance)
369{
370 size_t num_bins = static_cast<size_t>(std::ceil(max_distance / bin_width));
371 std::vector<T> radii(num_bins);
372 std::vector<T> g(num_bins, T(0));
373
374 // Precompute bin centers
375 for (size_t i = 0; i < num_bins; ++i)
376 radii[i] = (i + T(0.5)) * bin_width;
377
378 // Compute volume of domain
379 T volume = T(1);
380 for (const auto &range : axis_ranges)
381 volume *= (range.second - range.first);
382
383 size_t n_points = points.size();
384 T density = static_cast<T>(n_points) / volume;
385
386 // Histogram of pair distances
387 for (size_t i = 0; i < n_points; ++i)
388 {
389 for (size_t j = i + 1; j < n_points; ++j)
390 {
391 T dist = distance(points[i], points[j]);
392 if (dist < max_distance)
393 {
394 size_t bin = static_cast<size_t>(dist / bin_width);
395 if (bin < num_bins)
396 g[bin] += T(2); // count both i→j and
397 // j→i
398 }
399 }
400 }
401
402 // Normalize
403 T norm_factor = (T)n_points * density;
404
405 // Volume of spherical shell between r1 and r2 in N dimensions
406 auto sphere_volume = [](T radius)
407 { return std::pow(M_PI, N / 2.0) / std::tgamma(N / 2.0 + 1.0) * std::pow(radius, N); };
408
409 for (size_t i = 0; i < num_bins; ++i)
410 {
411 T r_inner = i * bin_width;
412 T r_outer = (i + 1) * bin_width;
413
414 // Volume of spherical shell
415 T shell_vol = sphere_volume(r_outer) - sphere_volume(r_inner);
416 g[i] /= norm_factor * shell_vol;
417 }
418
419 return {radii, g};
420}
