41template <typename T, size_t N>
42std::vector<T> distance_to_boundary(const std::vector<Point<T, N>> &points,
43 const std::array<std::pair<T, T>, N> &axis_ranges)
44{
45 std::vector<T> distances;
46 distances.reserve(points.size());
47
48 for (const auto &p : points)
49 {
50 // Find min distance to a boundary plane
51 T min_dist = std::numeric_limits<T>::max();
52
53 for (size_t d = 0; d < N; ++d)
54 {
55 T dist_to_min = std::abs(p[d] - axis_ranges[d].first);
56 T dist_to_max = std::abs(axis_ranges[d].second - p[d]);
57
58 min_dist = std::min({min_dist, dist_to_min, dist_to_max});
59 }
60
61 distances.push_back(min_dist);
62 }
63
64 return distances;
65}

106template <typename T, size_t N>
107std::vector<T> first_neighbor_distance(std::vector<Point<T, N>> &points)
108{
109 PointCloudAdaptor<T, N> adaptor(points);
110 KDTree<T, N> index(N, adaptor);
111 index.buildIndex();
112
113 std::vector<T> distance_sq;
114 distance_sq.reserve(points.size());
115
116 // first neighbor search only
117 const size_t k_neighbors = 1;
118
119 for (size_t i = 0; i < points.size(); ++i)
120 {
121 const auto &p = points[i];
122
123 std::vector<size_t> ret_indexes(k_neighbors + 1);
124 std::vector<T> out_dists_sqr(k_neighbors + 1);
125
126 nanoflann::KNNResultSet<T> result_set(k_neighbors + 1);
127 result_set.init(ret_indexes.data(), out_dists_sqr.data());
128
129 std::array<T, N> query;
130 for (size_t d = 0; d < N; ++d)
131 query[d] = p[d];
132
133 index.findNeighbors(result_set, query.data(), nanoflann::SearchParameters());
134
135 T dist_sq = 0.f;
136 if (ret_indexes.size() > 1)
137 {
138 const auto &q = points[ret_indexes[1]];
139 auto delta = p - q;
140 dist_sq = length_squared(delta) + T(1e-6);
141 }
142
143 distance_sq.push_back(dist_sq);
144 }
145
146 return distance_sq;
147}

198 std::vector<T> out_dists_sqr(k_neighbors + 1);

201 result_set.init(ret_indexes.data(), out_dists_sqr.data());
202
203 std::array<T, N> query;
204 for (size_t d = 0; d < N; ++d)
205 query[d] = p[d];
206
207 index.findNeighbors(result_set, query.data(), nanoflann::SearchParameters());
208
209 // Skip self at index 0

211 }
